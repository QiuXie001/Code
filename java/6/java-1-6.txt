package com.v1;

import com.v2.Student;
class Person{
	String name;
	
	public Person(String name) {
		super();
		this.name = name;
	}

	public void work() {
		System.out.println("我是父类的work方法");
	}
}
class Worker extends Person{
	String skill;
	public Worker(String name,String skill) {
		super(name);
		this.skill=skill;
	}
	public void work() {
		System.out.println("我是"+name+",我的工种是"+skill);
	}
	public void exercise() {
		System.out.println("业余我喜欢打篮球");
	}
}
class Teacher extends Person{
	String subject;
	public Teacher(String name,String subject) {
		super(name);
		this.subject=subject;
	}
	public void work() {
		System.out.println("我教的是"+subject);
	}
	public void study() {
		System.out.println("执行work");
		super.work();
		System.out.println("终身学习");
	}
}
public class MainClass {

public static void main(String[] arg) {
	Teacher teacher=new Teacher("张三","计算机");
	teacher.work();
	teacher.study();
	Worker work=new Worker("李四","车工");
	work.work();
	work.exercise();
	Person per=new Person("王五");
	per=teacher;
	per.work();
	per=work;
	per.work();
	Person per1=new Person("周仓");

}
}
