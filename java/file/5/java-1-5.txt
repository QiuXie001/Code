package com.v1;
class Student1{
	String name;
	int age;

	public Student1(String name,int age) {
		this(name);
		this.age=age;
	}
	public void speak() {
		System.out.println(name);
		System.out.println(age);
	}
	public String getName() {
		return name;
	}
	public void setName(String name) {
		this.name = name;
	}
	public int getAge() {
		return age;
	}
	public void setAge(int age) {
		this.age = age;
	}
	public Student1() {
		System.out.println("无参数构造方法");
	}
	public Student1(String name) {
		System.out.println("单参数构造方法");
		this.name=name;
	}
}
public class MainClass {
public static void main(String[] arg) {
	Student1 stu1=new Student1("正义",36);
	stu1.speak();
	Student1 stu2=new Student1();
	stu2.setName("温良");
	stu2.setAge(37);
	stu2.speak();
	System.out.println(stu2.getName());
	Student1 stu3=new Student1("周易");
	stu3.speak();
	
}
}

package com.v2;

import java.util.Scanner;

public class Student1 {

	String name;
	int age;
	public Student1(String name, int age) {
		super();
		this.name = name;
		this.age = age;
	}
	
	public String getName() {
		return name;
	}

	public void setName(String name) {
		this.name = name;
	}

	public int getAge() {
		return age;
	}

	public void setAge(int age) {
		if(age>0) {
			this.age = age;
		}
		else
		{
			System.out.println("年龄不能为负数,重新输入");
			Scanner scan=new Scanner(System.in);
			int newAge=scan.nextInt();
			setAge(newAge);
		}
	}
	public Student1() {
		
	}

	public void speak() {
		System.out.println(name);
		System.out.println(age);
	}
}
